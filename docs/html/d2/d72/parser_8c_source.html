570 static parsing_tree_st *generate_stmt_list(link_list_st *token_list, symbol_table_st *symbol_table) {
571  parsing_tree_st *root = NULL;
572  parsing_tree_st *stmt_list_tree_node = parsing_tree_new("stmt_list", NULL);
573  root = stmt_list_tree_node;
574  while (1) {
575  parsing_tree_st *stmt = generate_stmt(token_list, symbol_table);
576  parsing_tree_set_child(stmt_list_tree_node, stmt);
577 
578  link_node_st *semicolon_node = link_list_pop(token_list);
579  char *node_data = link_node_get_data(semicolon_node);
580  if (symbol_table_lookup(symbol_table, node_data) != DELIMITER)
581  raise_syntax_error(__LINE__, "expected: DELIMITER");
582  link_node_free(semicolon_node);
583  parsing_tree_st *semicolon = parsing_tree_new(";", NULL);
584  parsing_tree_set_sibling(stmt, semicolon);
585 
586  if (link_list_top(token_list) == NULL ||
587  symbol_table_lookup(symbol_table,
588  link_node_get_data(
589  link_list_top(token_list))) == CLOSE_CURLY_BRACKETS)
590  break;
591  stmt_list_tree_node = parsing_tree_new("stmt_list", NULL);
592  parsing_tree_set_sibling(semicolon, stmt_list_tree_node);
593  }
594  return root;
