267 static int s_load_program(const char *file_path, instruction_set_st *instructions) {
268  FILE *fin = NULL;
269  char input_buff[BUFFER_SIZE];
270  char *op_code;
271  char *op_first;
272  char *op_second;
273  int count = 0;
274 
275  if (file_path == NULL || instructions == NULL)
276  exit(EINVAL);
277 
278  fin = fopen(file_path, "r");
279  if (!fin) {
280  printf("load %s failed!\n", file_path);
281  exit(ENOENT);
282  }
283 
284  while (fgets(input_buff, BUFFER_SIZE, fin)) {
285 #ifdef DEBUG
286  fprintf(stderr, "line: %d, content: %s\n", count, input_buff);
287 #endif
288  if (count >= instructions->count) {
289  instructions->instructs = realloc(instructions->instructs,
290  instructions->count * RESIZE_FACTOR);
291 
292  if (instructions->instructs == NULL)
293  exit(ENOMEM);
294 
295  instructions->count *= RESIZE_FACTOR;
296  }
297 
298  // spilit by ' ', get first op code or label.
299  op_code = strtok(input_buff, " \t\r\n");
300 
301  // skip empty lines
302  if (op_code == NULL)
303  continue;
304 
305  // check op_code is label and put into label table with count(address).
306  if (strrchr(op_code, ':') != NULL) {
307 #ifdef DEBUG
308  fprintf(stderr, "label: %s\n", op_code);
309 #endif
310  s_insert_label(instructions->labels, op_code, count);
311  }
312  // spilit by ' ', get first oprand and second oprand.
313  op_first = strtok(NULL, " \t\r\n");
314  op_second = strtok(NULL, " \t\r\n");
315 
316 #ifdef DEBUG
317  fprintf(stderr, "code: %s, first %s, second %s\n", op_code, op_first, op_second );
318 #endif
319  instructions->instructs[count].op_code = strdup(op_code);
320  if (op_first != NULL)
321  instructions->instructs[count].op_first = strdup(op_first);
322  else
323  instructions->instructs[count].op_first = NULL;
324  if (op_second != NULL)
325  instructions->instructs[count].op_second = strdup(op_second);
326  else
327  instructions->instructs[count].op_second = NULL;
328  count++;
329  }
330  fclose(fin);
331 
332  return count;
333 }
