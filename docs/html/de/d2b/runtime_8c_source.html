310 static void s_evaluate(instruction_set_st *instructions) {
311  instruction_st *next_inst = NULL;
312  char *op_code = NULL;
313  char *op_first = NULL;
314  char *op_second = NULL;
315  int i;
316 
317  if (instructions == NULL)
318  exit(EINVAL);
319 
320  while ((next_inst = instruction_set_get_instruction(instructions)) != NULL) {
321  op_code = instruction_get_op_code(next_inst);
322  op_first = instruction_get_op_first(next_inst);
323  op_second = instruction_get_op_second(next_inst);
324 #ifdef DEBUG
325  fprintf(stderr, "code: %s, first: %s, second: %s\n", op_code, op_first, op_second );
326 #endif
327  if (op_code == NULL) {
328  fprintf(stderr, "Fetch failed\n");
329  exit(EINVAL);
330  }
331  // find method for op_code
332  i = 0;
333  while (g_operations[i].op_code != NULL) {
334  if (strcmp(op_code, g_operations[i].op_code) == 0) {
335  g_operations[i].eval_func(op_first, op_second);
336  break;
337  }
338  i++;
339  }
340 
341  if (g_operations[i].op_code == NULL) {
342 #ifdef DEBUG
343  fprintf(stderr, "label change scope\n");
344 #endif
345  // label change scope.(for1: scope++, for1_end: scope--)
346  if (strstr(op_code, "_end:"))
347  machine_memory_close_scope(s_machine_store);
348  else
349  machine_memory_open_scope(s_machine_store);
350  }
351  }
